$z=0 \qquad \frac{1}{x^{2}+y^{2}}=0 \implies \text{решений нет}$
$z=1 \qquad \frac{1}{x^{2}+y^{2}}=1 \implies x^{2}+y^{2}=1$
$z=4\qquad \frac{1}{x^{2}+y^{2}}=4 \implies x^{2}+y^{2}=\frac{1}{4}$
$z=-1\qquad \frac{1}{x^{2}+y^{2}}=-1 \implies \text{решений нет}$

1.2  $z=\frac{2x-y-1}{x+y-2}$
$$
\begin{aligned}
 & z=0 \\

 & \begin{cases}
x+y-2\ne 0 \\
2x-y-1=0
\end{cases} \implies \begin{cases}
y\neq2-x \\
y=2x-1
\end{cases}
\end{aligned}
$$

$$
\begin{aligned}
 & z=1 \\
 & \frac{2x-y-1}{x+y-2}=1 \\
 & 2x-y-1=x+y-2 \\
 & 2y=x+1 \\
 & y=\frac{x+1}{2}
\end{aligned}
$$

1.4  $z=x^{2}y+x$
$$
\begin{aligned}
 & z=0 \\
 & x^{2}y+x=0 \\
 & x(xy+1)=0 \\
 & \left[ \begin{aligned}
 & x=0 \\
 & xy+1=0
\end{aligned} \right. \implies \left[ \begin{aligned}
 & x=0 \\
 & y=-\frac{1}{x}
\end{aligned} \right.
\end{aligned}
$$
$$
\begin{aligned}
 & z=1 \\
 & \begin{aligned}
x^{2}y+x & =1 \\
xy+1 & =\frac{1}{x} \\
xy & =\frac{1}{x}-1 \\
y & =\frac{\frac{1}{x}-1}{x} \\
y & =\frac{1}{x^{2}}-\frac{1}{x} \\
y & =\frac{1-x}{x^{2}}
\end{aligned}
\end{aligned}
$$

3.1  $z=x^{y}$
частные производные первого порядка:
$\frac{ \partial z }{ \partial x }=yx^{y-1}$
$\frac{ \partial z }{ \partial y }=x^{y}\cdot \ln x$
частные производные второго порядка:
$$
\begin{aligned}
\frac{ \partial^{2}z }{ \partial x^{2} } =\frac{ \partial  }{ \partial x }\left( \frac{ \partial z }{ \partial x } \right)  & =z_{x x}'' \\[4pt]
 & =(yx^{y-1})'_{x} \\[4pt]
 & =y(y-1)x^{y-2}
\end{aligned}
$$
$$
\begin{aligned}
z_{y y}'' & =(x^{y} \cdot \ln x)'_{y} \\[4pt]
 & = x^{y}\cdot \ln ^{2}x
\end{aligned}
$$
смешанные производные:
$$
\begin{aligned}
\frac{ \partial  }{ \partial y }\left( \frac{ \partial z }{ \partial x } \right) & =z_{xy}'' \\[4pt]
 & =(yx^{y-1})_{y}' \\[4pt]
 & =x^{y-1}+y\cdot x^{y-1}\cdot \ln x
\end{aligned}
$$
$$
\begin{aligned}
z_{yx}'' & =(x^{y} \cdot \ln x)'_{x} \\
 & =y\cdot x^{y-1} \cdot \ln x + x^{y} \cdot \frac{1}{x} \\
 & =y\cdot x^{y-1} \cdot \ln x + x^{y-1}
\end{aligned}
$$
↑ то есть $z_{xy}''=z_{yx}''$
